# dbpsbp 'calc_bp_class' 
# 5

def calc_bp_class(dbp, sbp):
    # :
    # - dbp 80   sbp 120  1 
    # - dbp 85   sbp 130  1 
    # - dbp 90   sbp 140  1 
    # - dbp 100   sbp 160  1 
    # - dbp 110   sbp 180  1 
    #  1 
    return ((dbp >= 80) | (sbp >= 120)) * 1 \
         + ((dbp >= 90) | (sbp >= 140)) * 1 \
         + ((dbp >= 100) | (sbp >= 160)) * 1 \
         + ((dbp >= 110) | (sbp >= 180)) * 1
